   concept_open_loop_system;
   concept_closed_loop_system
};;

concept_signal
=> nrel_subdividing: {
   concept_reference_signal;
   concept_error_signal;
   concept_control_signal;
   concept_output_signal;
   concept_disturbance_signal;
   concept_measurement_signal
};;

   concept_state_space_controller

   nrel_positive_feedback

concept_frequency_response

   nrel_bandwidth;

   nrel_amplitude_margin
};;
